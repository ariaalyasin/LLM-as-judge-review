\section{Types of LLM Judges}

\begin{frame}{How LLMs Evaluate (The Proposed Solution)}
  \begin{itemize}
    \item \textbf{Goal:} Use strong LLMs (like GPT-4) as a \textbf{scalable, explainable surrogate} for costly human evaluation.
  \end{itemize}
  \vspace{0.2cm}
  \textbf{Three Judge Types:} 
  \begin{enumerate}
      \item \textbf{Pairwise Comparison:} The LLM is presented with a prompt and two models' answers, then asked to determine which is better (like Chatbot Arena).
      \item \textbf{Single Answer Grading:} The LLM assigns a direct score (e.g., 1-10) to a single model's response.
      \item \textbf{Reference-Guided Grading:} A baseline "gold standard" reference answer is provided to the LLM judge to assist in evaluation.
  \end{enumerate}
\end{frame}

\section{Limitations of LLM Judges}

\begin{frame}{Unmasking the Biases}
  While powerful, LLM Judges exhibit several inherent flaws:
  \begin{itemize}
    \item \textbf{Position Bias:} A tendency to favor the response placed in the first position.
    \item \textbf{Verbosity Bias:} Favoring longer, wordy answers, even if they aren't necessarily better.
    \item \textbf{Self-Enhancement Bias:} Models tend to rate their own generated answers higher.
    \item \textbf{Limited Reasoning:} Struggling to accurately grade complex math and logic without help.
  \end{itemize}
\end{frame}

\begin{frame}{Quantifying the Flaws: Position and Verbosity Bias}
   \begin{figure}

    \item GPT-3.5 and Claude-v1 heavily favor the \textit{first} answer presented.
    \item Models are also vulnerable to attacks where appending a repetitive, meaningless list of words to an answer artificially inflates its win rate due to \textbf{verbosity bias}.
  \end{itemize}
\end{frame}

\section{Addressing the Biases}

\begin{frame}{Mitigating LLM Judge Limitations}
  To use LLMs as judges reliably, these strategies must be implemented:
  \begin{itemize}
    \item \textbf{Swapping Positions:} Run the comparison twice ($A$ vs $B$, then $B$ vs $A$) to cancel out position bias.
    \item \textbf{Few-Shot \& Chain-of-Thought (CoT):} Forcing the judge to reason step-by-step \textit{before} scoring.
    \item \textbf{Reference-Guided Grading:} Providing ground-truth facts.
    \item \textbf{Fine tuning the model}.
  \end{itemize}
\end{frame}

\section{Introduction \& The Problem}

\begin{frame}{The Evaluation Paradigm}
  \begin{itemize}
    \item \textbf{The Shift:} As LLMs advance, evaluating their alignment with human intent is critical.
    \item \textbf{The Solution:} Researchers increasingly use LLMs like GPT-4 as automated referees to score and compare candidate models due to high human annotation costs.
    \item \textbf{The Unknown:} It remains unclear how reliable LLMs actually are as evaluators against perturbations.
  \end{itemize}
\end{frame}

\section{The Core Discovery: Positional Bias}

\begin{frame}{The Achilles' Heel of LLM Evaluators}
  \begin{itemize}
    \item \textbf{Positional Bias:} The LLM-as-evaluator paradigm has a severe flaw. Quality ranking can be hacked by altering the order of candidates.
    \item \textbf{Model Preferences:} GPT-4 consistently prefers the \textit{first} displayed candidate, while ChatGPT strongly favors the \textit{second}.
    \item \textbf{The Flip:} Merely swapping the presentation order can completely reverse the evaluation outcomes.
  \end{itemize}
\end{frame}

\begin{frame}{Visualizing the Bias}
  \begin{figure}
      \centering
      \includegraphics[width=0.8\textwidth]{images/4/figure1.png} 
      \caption{Swapping order overturns comparison results.}
  \end{figure}
\end{frame}

\begin{frame}{Quantifying the Unfairness}
  \begin{itemize}
    \item \textbf{Conflict Rate:} A metric introduced to measure the proportion of conflicting results given by the evaluator when candidate positions are swapped.
    \item \textbf{The Formula:}
    $$ \text{Conflict Rate} = \frac{\text{Number of Conflicting Evaluations}}{\text{Total Number of Evaluations}} $$
    \item \textbf{Extreme Volatility:} ChatGPT exhibits a massive conflict rate of up to 82.5\% when comparing Vicuna-13B to ChatGPT.
  \end{itemize}
\end{frame}

\begin{frame}{Win Rate Fluctuation Data}
  \begin{figure}
      \centering
      \includegraphics[width=1\textwidth]{images/4/table2.png}
      \caption{Win rate fluctuation and Conflict Rates across different models.}
  \end{figure}
\end{frame}

\begin{frame}{When Does Bias Hit the Hardest?}
  \begin{itemize}
    \item The degree of positional bias correlates directly with the quality difference between the two responses.
    \item \textbf{Small Gap:} When candidate quality is very similar, the LLM is significantly more likely to produce conflicting results based purely on position.
    \item \textbf{Large Gap:} If one response is obviously superior, the bias has less impact.
  \end{itemize}
\end{frame}

\begin{frame}{Conflict Rate vs. Score Gap}
  \begin{figure}
      \centering
      \includegraphics[width=0.7\textwidth]{images/4/figure2.png}
      \caption{The conflict rate is negatively correlated with the score gap.}
  \end{figure}
\end{frame}

\section{The Calibration Framework}

\begin{frame}{Fixing the System: The 3-Step Strategy}
  A calibration framework with three strategies is proposed to neutralize positional bias:
  \begin{itemize}
    \item \textbf{1. Multiple Evidence Calibration (MEC):} Forces the auto-regressive LLM to generate evaluation evidence \textit{before} assigning a score, averaging multiple samples for stability.
    \item \textbf{2. Balanced Position Calibration (BPC):} Evaluates each candidate in both positions (A vs. B, then B vs. A) and averages the final scores.
    \item \textbf{3. Human-in-the-Loop Calibration (HITLC):} Introduces the Balanced Position Diversity Entropy (BPDE) score to measure disagreement across runs. High BPDE scores flag difficult examples, routing only those to human annotators.
  \end{itemize}
\end{frame}

\begin{frame}{The Full Calibration Pipeline}
  \begin{figure}
      \centering
      \includegraphics[width=0.9\textwidth]{images/4/figure3.png}
      \caption{Demonstration of the calibration framework with MEC, BPC, and HITLC.}
  \end{figure}
\end{frame}

\begin{frame}{Optimizing the Calibration Parameters}
  To make MEC and BPC viable, hyperparameter tuning is critical:
  \begin{itemize}
    \item \textbf{Number of Evidences ($k$):} Sampling multiple explanations stabilizes the evaluation.
    \begin{itemize}
        \item Empirical results show $\mathbf{k=3}$ is the optimal sweet spot, maximizing accuracy while controlling API costs.
    \end{itemize}
    \vspace{0.3cm}
    \item \textbf{Sampling Temperature ($t$):} 
    \begin{itemize}
        \item \textit{Low temp ($t=0.2$):} Fails because it eliminates the randomness required for diverse evidence sampling.
        \item \textit{High temp ($t=1.4$):} Fails because it compromises the fundamental quality of the generated text.
        \item \textit{Optimal range:} $\mathbf{t \in [0.6, 1.0]}$ yields the best alignment.
    \end{itemize}
  \end{itemize}
\end{frame}

\section{Results \& Impact}

\begin{frame}{Results \& Impact}
  \begin{itemize}
    \item \textbf{Alignment Boost:} MEC and BPC strategies enhanced the evaluation alignment accuracy with human judgment by 9.8\% for GPT-4 and 14.3\% for ChatGPT.
    \item \textbf{Efficiency:} Utilizing HITLC with a 20\% human annotation threshold allows ChatGPT to achieve comparable alignment to average human performance.
    \item \textbf{Cost Reduction:} This strategic routing method effectively reduces human annotation costs by up to 39\%.
  \end{itemize}
\end{frame}

\section{Conclusion}

\begin{frame}{Summary}
  \begin{itemize}
    \item \textbf{The Illusion of Fairness:} Blindly trusting LLMs as evaluators leads to systematically biased results due to severe positional sensitivity.
    \item \textbf{The Solution:} We must implement rigorous calibration protocols to ensure robust evaluation metrics.

\section{Quantifying Bias Sources (The CALM Framework)}

\begin{frame}{The CALM Pipeline}
  \textbf{Comprehensive Assessment of Language Model Judge Biases}
  
  \vspace{0.3cm}
  \begin{itemize}
      \item \textbf{Attack-and-Detect Approach:} Automatically injects perturbations into instructions or responses without altering the actual correctness of the answer.
      \item \textbf{Key Advantage:} Completely eliminates subjective human assessment, allowing for objective and scalable evaluation across massive datasets.
  \end{itemize}
  \vspace{0.5cm}
  \textit{(See the architectural pipeline on the next slide)}
\end{frame}

\begin{frame}{Figure: CALM Framework Architecture}
  \begin{figure}[h]
      \centering
      \includegraphics[width=0.95\textwidth]{images/3/pipeline.png} 
      \caption{The automated pipeline for bias assessment.}
  \end{figure}
\end{frame}

\begin{frame}{Bias Assessment Problem Formulation}
  To systematically test the judge, we formulate the input prompt $P$:
  \vspace{0.3cm}
  
  \textbf{Standard Scoring:}
  \begin{itemize}
      \item Base prompt: $P = (I, Q, R)$ 
      \item \textit{Where $I$ is Instruction, $Q$ is Question, $R$ is Response.}
  \end{itemize}

  \vspace{0.3cm}
  \textbf{Pairwise Comparison (Targeting the Bias):}
  \begin{itemize}
      \item We compare two answers: $P = (I, Q, R_1, R_2)$
      \item We apply an automated perturbation function $g(\cdot)$ to one answer.
      \item \alert{The Hacked Prompt:} $\hat{P} = (I, Q, R_1, g(R_2))$
  \end{itemize}
\end{frame}

\begin{frame}{Bias Types \& Automated Perturbation}
  \textbf{How $g(\cdot)$ works:}
  \begin{itemize}
      \item The perturbation function uses a principle-guided LLM (like Constitutional AI) to deliberately inject specific biases.
      \item \textbf{The Rule:} It changes the \textit{style, length, or context} of the answer, but \alert{never} changes the fundamental factual correctness.
  \end{itemize}

  \vspace{0.5cm}
  \textbf{The 12 Bias Sources Assessed:}
  \begin{itemize}
      \item \textbf{Formatting:} Position, Verbosity, Distraction
      \item \textbf{Logical:} Fallacy-Oversight, Authority
      \item \textbf{Social:} Compassion-Fade, Bandwagon, Sentiment, Diversity
      \item \textbf{Behavioral:} Chain-of-Thought, Self-Enhancement, Refinement
  \end{itemize}
\end{frame}

\begin{frame}{Figure: The 12 Types of Biases}
  \begin{figure}[h]
      \centering
      \includegraphics[width=0.9\textwidth]{images/3/bias_table.png} 
      \caption{Definitions and examples of the 12 biases evaluated.}
  \end{figure}
\end{frame}

\begin{frame}{Evaluation Datasets}
  To ensure comprehensive testing, CALM utilizes three distinct dataset categories:
  
  \vspace{0.5cm}
  \begin{enumerate}
      \item \textbf{Fact-Related Datasets:} 
      \begin{itemize}
          \item Focuses on Math and Science (e.g., GSM8K, MATH). Answers have a clear objective truth.
      \end{itemize}
      \item \textbf{Refinement-Aware Datasets:} 
      \begin{itemize}
          \item Humanities and open-ended QA where answers are improved via conversational history.
      \end{itemize}
      \item \textbf{Alignment Datasets:} 
      \begin{itemize}
          \item Data based on subjective human preference (DPO datasets). Quality differences between answers are much subtler.
      \end{itemize}
  \end{enumerate}
\end{frame}

\section{Quantitative Evaluation Metrics}

\begin{frame}{Evaluation Metrics (How we measure fairness)}
  Instead of relying on human intuition, CALM calculates specific metrics by running the judge multiple times:

  \vspace{0.3cm}
  \begin{itemize}
      \item \textbf{Robustness Rate (RR):} Measures how consistently the model makes the \textit{exact same decision} before and after a bias is injected. Higher means better resistance.
      \item \textbf{Consistency Rate (CR):} Evaluates stability. Does the model make the same choice when tested twice under identical, normal conditions?
      \item \textbf{Accuracy (Original vs. Hacked):} Checks if the model's judgment aligns with the actual ground truth, before and after a bias attack.
      \item \textbf{Error Rate:} Specifically used to quantify severe behavioral biases (like favoring its own outputs or favoring edited text).
  \end{itemize}
\end{frame}

\section{Key Results \& Findings}

\begin{frame}{Overall Model Vulnerabilities}
  \textbf{Big Picture Findings:}
  \vspace{0.3cm}
  \begin{itemize}
      \item No model is perfectly immune. Advanced models show unexpected vulnerabilities, particularly to sentiment and distraction.
      \item \textbf{Claude-3.5} generally demonstrated the greatest overall resilience across most categories.
      \item Performance varies wildly depending on the specific type of bias injected.
  \end{itemize}
  \vspace{0.5cm}
  \textit{(See the radar chart comparison on the next slide)}
\end{frame}

\begin{frame}{Figure: Robustness Rate Radar Charts}
  \begin{figure}[h]
      \centering
      \includegraphics[width=0.9\textwidth]{images/3/results_radar.png} 
      \caption{Overall robustness rates of models against the 12 biases.}
  \end{figure}
\end{frame}

\begin{frame}{Summary of Key Findings (1/2)}
  \begin{itemize}
      \item \textbf{Even advanced models can exhibit unexpected vulnerabilities in judgment.} Top-tier models are not a silver bullet for fair evaluation.
      \item \textbf{Bias is more pronounced in the alignment dataset compared to the fact-related dataset.} Subjective tasks are much easier to disrupt.
      \item \textbf{Position bias increases with more answer candidates.} Evaluating 3 or 4 options drastically reduces the judge's robustness.
      \item \textbf{Avoid using the same model to generate and judge answers.} Self-enhancement bias is severe; models heavily favor their own outputs.
  \end{itemize}
\end{frame}

\begin{frame}{Summary of Key Findings (2/2)}
  \begin{itemize}
      \item \textbf{Response length influences model judgment in complex ways.} Some favor verbosity blindly, others penalize it.
      \item \textbf{Different types of fake authorities interfere to varying degrees.} Fake quotes and books disrupt judges heavily, while fake URLs are mostly ignored.
      \item \textbf{LLMs show sensitivity to irrelevant content and emotional elements.} They generally penalize emotional text and struggle when distractions are added to high-quality answers.
      \item \textbf{Explicit introduction of minority groups influences choices.} Models react differently to exposed demographic identities.
      \item \textbf{Chain-of-Thought (CoT) improves LLMs evaluation accuracy.} Forcing step-by-step reasoning actively guards against bias.
  \end{itemize}
\end{frame}

\section{Conclusion}

\begin{frame}{Recommendations for Deployment}
  \textbf{How to build reliable LLM Judges:}
  \vspace{0.3cm}
  \begin{enumerate}
      \item \textbf{Enforce Advanced Reasoning:} Use Chain-of-Thought (CoT) prompting to force logic before a final score is given.
      \item \textbf{Implement Prompt Safeguards:} Explicitly instruct the model to ignore formatting, identity markers, and emotional tone within the system prompt.
      \item \textbf{Proactive Detection:} Utilize automated frameworks (like CALM) to detect vulnerabilities in evaluation templates \alert{before} deploying the system.
  \end{enumerate}
\end{frame}

\section{Motivation \& Core Problem}

\begin{frame}{Why Do We Need JudgeLM?}
	\begin{itemize}
		\item \textbf{The Problem:} Evaluating LLMs in open-ended tasks is difficult because traditional metrics like BLEU and ROUGE cannot fully capture answer quality.
		\item \textbf{Human Evaluation:} Reliable, but slow, expensive, and hard to scale.
		\item \textbf{GPT-4 as Judge:} Strong, but depends on closed APIs, can be expensive, and may create privacy and reproducibility concerns.
		\item \textbf{JudgeLM's Goal:} Fine-tune open-source LLMs to become scalable, efficient, and reliable judges.
	\end{itemize}
\end{frame}

\begin{frame}{Main Idea of JudgeLM}
	\begin{itemize}
		\item \textbf{Core Proposal:} Train LLMs themselves to act as judges for open-ended evaluation.
		\item \textbf{Model Sizes:} JudgeLM is trained at 7B, 13B, and 33B parameters.
		\item \textbf{Teacher Judge:} GPT-4 is used to generate high-quality judgment labels.
		\item \textbf{Output Style:} The judge gives scores and detailed reasons for answer pairs.
	\end{itemize}
\end{frame}

	\begin{figure}

\end{frame}

\section{JudgeLM Dataset \& Training Pipeline}

\begin{frame}{Dataset Construction}
	\begin{itemize}
		\item \textbf{Seed Tasks:} The authors collect 105K instruction tasks from sources such as Alpaca-GPT4, Dolly-15K, GPT4All-LAION, and ShareGPT.
		\item \textbf{Candidate Answers:} Answers are generated by 11 open-source LLMs, including LLaMA, Alpaca, Vicuna, Koala, and others.
		\item \textbf{Teacher Labels:} GPT-4 judges randomly sampled answer pairs and provides scores plus explanations.
		\item \textbf{Split:} 100K samples are used for training, and 5K are used for validation.
	\end{itemize}
\end{frame}

\begin{frame}{Why Traditional Metrics Fail}
	\begin{figure}
		\centering
		\includegraphics[width=0.9\textwidth]{images/5/figure2.png}
		\caption{Traditional metrics can prefer incomplete answers, while LLM judges can reason about quality.}
	\end{figure}
\end{frame}

\begin{frame}{Training and Evaluation Format}
	\begin{itemize}
		\item \textbf{Input:} A task, two candidate answers, and sometimes a reference answer.
		\item \textbf{Output:} Scores for both answers and a reason explaining the judgment.
		\item \textbf{Judgment Logic:}
		\begin{enumerate}
			\item If Answer 1 gets a higher score, Answer 1 wins.
			\item If Answer 2 gets a higher score, Answer 2 wins.
			\item If both scores are equal, it is a tie.
		\end{enumerate}
		\item \textbf{Efficiency Trick:} Reason generation can be skipped when only the final judgment is needed.
	\end{itemize}
\end{frame}

\section{Biases in Fine-tuned Judges}

\begin{frame}{Three Key Biases}
	\begin{itemize}
		\item \textbf{1. Position Bias:} The judge may prefer the answer shown first, regardless of quality.
		\item \textbf{2. Knowledge Bias:} The judge may fail when its pre-trained knowledge is missing or wrong.
		\item \textbf{3. Format Bias:} The judge may perform well only in the same prompt format used during fine-tuning.
	\end{itemize}
\end{frame}

\section{Proposed Methods}

\begin{frame}{Method 1: Swap Augmentation}
	\begin{itemize}
		\item \textbf{Problem Addressed:} Position bias.
		\item \textbf{Idea:} During training, swap Answer 1 and Answer 2.
		\item \textbf{Label Adjustment:} The GPT-4 scores and answer indexes are swapped too, so the correct judgment stays consistent.
		\item \textbf{Effect:} Forces the judge to focus on answer content instead of answer position.
	\end{itemize}
\end{frame}

\begin{frame}{Method 2: Reference Support}
	\begin{itemize}
		\item \textbf{Problem Addressed:} Knowledge bias.
		\item \textbf{Idea:} Give the judge a reference answer during training.
		\item \textbf{Why It Helps:} The judge can rely on external truth instead of only its internal knowledge.
		\item \textbf{Extra Benefit:} Reference answers can guide the judge toward task-specific preferences.
	\end{itemize}
\end{frame}

\begin{frame}{Method 3: Reference Drop}
	\begin{itemize}
		\item \textbf{Problem Addressed:} Format bias.
		\item \textbf{Idea:} Randomly remove reference answers during training.
		\item \textbf{Why It Helps:} The model learns to judge both with and without references.
		\item \textbf{Result:} JudgeLM becomes more flexible and less dependent on one fixed prompt format.
	\end{itemize}
\end{frame}

\section{The Missing Perspective: System-Level Ranking}

\begin{frame}{The Gap in Existing Judge Benchmarks}
	\begin{itemize}
		\item \textbf{Current Practice:} LLM judges are commonly evaluated at the \textit{instance level}: can the judge correctly compare individual responses?
		\item \textbf{Real-World Goal:} Researchers often use these judgments to rank entire models, systems, or configurations.
		\item \textbf{The Problem:} A judge that performs well on individual responses does not necessarily produce an accurate overall system ranking.
		\item \textbf{Why This Matters:} Errors may be distributed unevenly across systems, creating unfair rankings even when average instance-level accuracy looks strong.
	\end{itemize}
\end{frame}

\begin{frame}{Instance-Level vs. System-Level Evaluation}
	\begin{figure}
		\centering
		\includegraphics[width=0.8\textwidth]{images/6/figure1.png}
		\caption{Instance-level judges compare individual responses, while system-level judges rank complete systems.}
	\end{figure}
\end{frame}

\begin{frame}{The Main Contribution: JuStRank}
	\begin{itemize}
		\item \textbf{JuStRank:} A benchmark designed specifically to evaluate judges as \textit{system rankers}.
		\item \textbf{Core Question:} Can an automated judge rank multiple LLM systems similarly to human evaluators?
		\item \textbf{Evaluation Principle:} Aggregate the judge's scores over many outputs, construct a model ranking, and compare it with a human-based gold ranking.
		\item \textbf{Broader Goal:} Help practitioners select the right judge for model comparison and model-selection decisions.
	\end{itemize}
\end{frame}

\section{JuStRank Task Formulation \& Pipeline}

\begin{frame}{System-Level Judge Pipeline}
	\begin{figure}
		\centering
		\includegraphics[width=0.95\textwidth]{images/6/figure2.png}
		\caption{Pipeline for generating system-level rankings from individual judge scores.}
	\end{figure}
\end{frame}

\begin{frame}{Task Formulation}
	\begin{itemize}
		\item Let $S = {s_l}{l=1}^{L}$ be the set of target systems and $I = {i_k}{k=1}^{K}$ be the user instructions.
		\item Each system $s_l$ generates a response $r_k^l$ for instruction $i_k$.
		\item A judge $j_p$ assigns a scalar quality score:
		$$ j_p(i_k, r_k^l) = \text{Score}_{k,l}^p $$
		\item An aggregation method converts the response-level scores into one score per system:
		$$ V^{p,a} = a(j_p(R)) $$
		\item The final system ranking is compared with a human-based gold ranking.
	\end{itemize}
\end{frame}

\section{Experimental Setup}

\begin{frame}{Experimental Setup}
	\begin{itemize}
		\item \textbf{System Responses:} Arena Hard v0.1 provides 500 challenging prompts and outputs from 63 systems.
		\item \textbf{Scale:} The benchmark evaluates approximately 32K instruction-response pairs.
		\item \textbf{Judges:} JuStRank studies 48 judge realizations:
		\begin{itemize}
			\item 8 reward-model judges.
			\item 40 generative LLM judges: 10 LLMs evaluated using 4 prompting styles.
		\end{itemize}
		\item \textbf{Total Judgments:} 1.5 million individual scores.
		\item \textbf{Gold Ranking:} Human preference data from the Chatbot Arena English Hard Prompts subset.
	\end{itemize}
\end{frame}

\begin{frame}{Four Ways to Use an LLM as a Judge}
	\begin{itemize}
		\item \textbf{1. Numeric:} Assign an absolute score between 0 and 100.
		\item \textbf{2. Likert:} Assign a textual rating: Very Bad, Bad, Mediocre, Good, or Very Good.
		\item \textbf{3. TokenProbs:} Answer whether a response is good, then use the probabilities of yes'' and no'' as a score.
		\item \textbf{4. Anchor:} Compare each system response against an anchor response from GPT-4-0314.
	\end{itemize}
	
	\vspace{0.3cm}
	\textbf{Aggregation Methods:}
	\begin{itemize}
		\item Win-Rate, Mean, Median, and Bradley-Terry (BT).
	\end{itemize}
\end{frame}

\section{Judge Behavior: Decisiveness \& Bias}

\begin{frame}{Emergent Judge Behavior: Decisiveness}
	\begin{itemize}
		\item \textbf{Decisiveness:} Some judges strongly amplify the performance gap between better and weaker systems.
		\item \textbf{Example:} When humans mildly prefer one system, a decisive judge may produce a much more extreme preference.
		\item \textbf{Potential Benefit:} Decisive judges can separate strong and weak models using fewer examples.
		\item \textbf{Caution:} Greater decisiveness does not always mean greater fairness or calibration.
	\end{itemize}
\end{frame}

\begin{frame}{System-Specific Bias}
	\begin{itemize}
		\item \textbf{Bias Definition:} A judge is biased toward a system if its predicted win rates are consistently higher or lower than the human-based win rates.
		\item \textbf{Positive Bias:} Some systems receive systematically inflated rankings.
		\item \textbf{Negative Bias:} Other systems are evaluated too harshly.
		\item \textbf{Example:} Many judges show positive bias toward Athene-70B, while GPT-4-0613 receives negative bias.
		\item \textbf{Core Risk:} A biased judge can distort model-selection decisions even when its average accuracy appears acceptable.
	\end{itemize}
\end{frame}

\section{Discussion \& Conclusion}

\begin{frame}{Discussion and Limitations}
	\begin{itemize}
		\item \textbf{Prompt Sensitivity:} Judge behavior can change when the realization prompt changes.
		\item \textbf{Human Preference Is Complex:} The gold ranking combines subjective preferences such as helpfulness, safety, style, and coherence.
		\item \textbf{Dataset Alignment:} Judge scores are collected on Arena Hard, while the gold ranking comes from Chatbot Arena's English Hard Prompts subset.
		\item \textbf{Future Work:} Train dedicated system-level judges, explore judge ensembles, and test additional aggregation methods.
	\end{itemize}
\end{frame}

\begin{frame}{Summary}
	\begin{itemize}
		\item \textbf{The Missing Evaluation Layer:} Instance-level judge quality does not guarantee accurate system rankings.
		\item \textbf{The Contribution:} JuStRank evaluates judges by comparing their model rankings with a human-based gold ranking.
		\item \textbf{Key Findings:}
		\begin{enumerate}
			\item Reward models can compete with much larger LLM judges.
			\item Prompting style and score aggregation strongly affect ranking quality.
			\item Judges exhibit emergent decisiveness and system-specific bias.
		\end{enumerate}
		\item \textbf{Final Takeaway:} The best judge depends on the actual goal. A judge used to select models should be evaluated as a system ranker, not only as an instance-level evaluator.
	\end{itemize}
\end{frame}
