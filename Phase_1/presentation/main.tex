$$ D_A = A^+ = \bar{A}B\bar{Y} + \bar{A}BY + A\bar{B}\bar{Y} + ABY = \bar{A}B + A(\bar{B}\bar{Y} + BY) = \bar{A}B + A\overline{(B \oplus Y)} $$

$$ D_B = B^+ = \bar{A}\bar{B}Y + \bar{A}BY = \bar{A}Y(\bar{B} + B) = \bar{A}Y $$

$$ Z = \bar{A}\bar{B}\bar{Y} + \bar{A}\bar{B}Y + AB\bar{Y} + ABY = \bar{A}\bar{B} + AB = \overline{(A \oplus B)} $$

\problem 

برای طراحی این فلیپ‌فلاپ از ساختار Master-Slave استفاده می‌کنیم. جهت اعمال سیگنال بارگذاری ($L$)، یک تابع مولتی‌پلکسری برای ورودی داده داخلی ($D_{in}$) تعریف می‌شود؛ به طوری که اگر $L=1$ باشد داده جدید ($D$) و اگر $L=0$ باشد، خروجی فعلی مدار ($Q$) جهت حفظ حالت انتخاب می‌شود:
$$ D_{in} = L \cdot D + \bar{L} \cdot Q $$

با توجه به محدودیت مسئله مبنی بر استفاده‌ی انحصاری از گیت‌های NAND و NOT، این رابطه را بر اساس قانون دمرگان بازنویسی می‌کنیم:
$$ D_{in} = \overline{ \overline{(L \cdot D)} \cdot \overline{(\bar{L} \cdot Q)} } $$

در ادامه، روابط تحریک لچ‌های مادری و پیرو \lr{($\bar{S}\bar{R}$ Latch)} بر پایه‌ی ورودی داده‌ی داخلی و سیگنال ساعت ($C$) به صورت زیر استخراج می‌شوند:

\noindent \textbf{طبقه Master (فعال در سطح پایین ساعت $C=0$):}
\begin{itemize}
    \item $\overline{S}_M = \overline{D_{in} \cdot \overline{C}}$
    \item $\overline{R}_M = \overline{\overline{D_{in}} \cdot \overline{C}}$
\end{itemize}

\noindent \textbf{طبقه Slave (فعال در سطح بالای ساعت $C=1$):}
\begin{itemize}
    \item $\overline{S}_S = \overline{Q_M \cdot C}$
    \item $\overline{R}_S = \overline{\overline{Q_M} \cdot C}$
\end{itemize}

در نهایت، خروجی‌های حاصل از لچ Slave به عنوان خروجی‌های نهایی مدار ($Q$ و $\overline{Q}$) در نظر گرفته شده و جهت حفظ پایداری وضعیت، به گیت‌های ورودی فیدبک داده می‌شوند.

\problem 

این مدار یک شمارنده دهدهی  است که دنباله مشخصی از حالت \lr{0000} تا \lr{1001} را طی می‌کند و رفتار آن برای حالات \lr{1010} تا \lr{1111} نامشخص \lr{(Don't Care)} فرض می‌شود. با فرض این که خروجی‌ها به صورت $Q_3Q_2Q_1Q_0$ باشند، معادلات بهینه‌ی سیستم به شرح زیر است:

\subsubsection*{1) پیاده‌سازی با فلیپ‌فلاپ‌های SR}

با دوقطبی کردن جدول انتقال وضعیت و اعمال جدول تحریک فلیپ‌فلاپ‌های SR، توابع ورودی پس از ساده‌سازی با جدول کارنو به دست می‌آیند:

\begin{itemize}
    \item $S_0 = \overline{Q_0}, \quad R_0 = Q_0$
    \item $S_1 = \overline{Q_3}\overline{Q_1}Q_0, \quad R_1 = Q_1Q_0$
    \item $S_2 = \overline{Q_2}Q_1Q_0, \quad R_2 = Q_2Q_1Q_0$
    \item $S_3 = Q_2Q_1Q_0, \quad R_3 = Q_3Q_0$
\end{itemize}

\subsubsection*{2) پیاده‌سازی با فلیپ‌فلاپ‌های T}

با استفاده از رابطه تحریک فلیپ‌فلاپ نوع T ($T = Q \oplus Q^+$) و ساده‌سازی خانه‌های جدول کارنو، توابع کنترل ورودی هر فلیپ‌فلاپ به صورت زیر حاصل می‌شوند:

\begin{itemize}
    \item $T_0 = 1$
    \item $T_1 = \overline{Q_3}Q_0$
    \item $T_2 = Q_1Q_0$
    \item $T_3 = Q_2Q_1Q_0 + Q_3Q_0$
\end{itemize}
